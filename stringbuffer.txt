package Phase1eLearningAssiignment1;

public class StringBuilderandBuffer {
	
public static void main(String[] args) {
		
		StringBuffer buffer = new StringBuffer("Hey whatsap?");
		buffer.append(".anything special");
		StringBuilder builder = new StringBuilder(" nothing builder");
		builder.append(".today you are looking good");
		System.out.println(buffer);
	}


}

nothing builder.today you are looking good-->builder  o/p
